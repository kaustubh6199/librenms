rsCPMaxCIDREntriesTuning OBJECT IDENTIFIER ::= { rsCP 27}

rsCPMaxCIDREntries OBJECT-TYPE
    SYNTAX INTEGER
    ACCESS  read-only
    STATUS  mandatory
    DESCRIPTION
      "Maximal number of entries in the CIDRs table."
    ::=  { rsCPMaxCIDREntriesTuning 1 }

rsCPMaxCIDREntriesAfterReset OBJECT-TYPE
    SYNTAX INTEGER(50..300)
    ACCESS  read-write
    STATUS  mandatory
    DESCRIPTION
      "Future maximal number of entries in the CIDRs table."
    ::=  { rsCPMaxCIDREntriesTuning 2 }

-- Application-specific traps for PeerDirector
-- Each enterprise-specific trap has two bound variables describing the exact nature of
-- the trap. rndErrorDesc provides a detailed description of the problem, including the
-- related interface number, board number or any other applicable variable.
-- rndErrorSeverity describes the severity of this problem.

-- Trap template
-- -------------
--    TRAP-TYPE
--       ENTERPRISE  rsServerDispatcher
--       VARIABLES   { rndErrorDesc, rndErrorSeverity }
--       DESCRIPTION
--             ""
--       ::=
-- -------------

rsPDGeneralError TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
         DESCRIPTION
             "This is a trap for all Peer Director's errors."
     ::=  1

rsPDInfoTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  2

rsPDWarningTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  3

rsPDErrorTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  4

rsPDFatalErrorTrap TRAP-TYPE
         ENTERPRISE rsCP
         VARIABLES { rndErrorDesc, rndErrorSeverity }
     ::=  5
